Step 6: percentiles

Estimates the percentile distribution of blood total mercury across age groups.

# Compute percentiles
lapply(
  X = levels(survey_design$variables$agegrp),
  FUN = function(x) {
    svyquantile(
      x = ~ Bmercury_CU,
      design = survey_design %>% 
        filter(agegrp == x),
      quantiles = c(0.25, 0.5, 0.75, 0.95),
      interval.type = "quantile",
      ties = "rounded", 
      na.rm = TRUE
    )[[1]]
  }
)
